001/** Class representing the interface between the processor
002 * and physical memory.
003 */
004public class InstructionMemory
005{
006    private final Memory m_memory;
007    
008    public InstructionMemory(Memory memory)
009    {
010        m_memory = memory;
011    }
012
013    public BitField32 fetch(BitField64 pc)
014    {
015        int address = pc.toIntSigned();
016        byte[] data = m_memory.read(address, 4);
017        return new BitField32(data);
018    }
019}
