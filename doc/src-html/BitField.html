001import java.util.Arrays;
002
003/**
004 * Class representing a bit-field of arbitrary (greater than zero)
005 * size.  There are a number of constructors for creating BitField
006 * objects from various other types.  There are a number of toXXX()
007 * methods for getting the value in different types.
008 */
009public class BitField
010{
011    private final boolean[] m_bits;
012
013    /**
014     * Constructor to create a bit-field of the specified
015     * size, with all bits set to zero.
016     *
017     * @param size value greater than zero indicating the number of
018     * bits to represent.
019     */
020    public BitField(int size)
021    {
022        if(size <= 0){
023            throw new IllegalArgumentException(String.format("BitField.BitField(%d) is not allowed.", size));
024        }
025        m_bits = new boolean [ size ];
026    }
027
028    /**
029     * Constructor to create a bit field out of a String containing
030     * '1' and '0' characters.  The MSB is assumed to be on the left
031     * side of the string (index zero), with the LSB on the right side
032     * of the string (length - 1).
033     *
034     * In the parsing logic, '0' characters are 'false' and all other
035     * characters are 'true'.
036     *
037     * @param value non-null, non-empty string to be converted to a
038     * byte array.
039     */
040    public BitField(String value)
041    {
042        if(null == value){
043            throw new IllegalArgumentException("BitField.BitField((String)null) is not allowed.");
044        }
045        if(value.length() == 0){
046            throw new IllegalArgumentException("BitField.BitField(\"\") is not allowed.");
047        }
048        m_bits = new boolean [ value.length() ];
049        // change the order
050        for(int i=value.length()-1, idx = 0; i >= 0; i--, idx++){
051            m_bits[idx] = !(value.charAt(i) == '0');
052        }
053    }
054
055    /**
056     * Constructor to create a bit-field from a byte array.  In the
057     * argument byte array, the LSB is (data[0] AND 0x80) and the MSB
058     * is (data[data.length-1] AND 0x1).
059     *
060     * @param data non-null, non-zero-length byte array that is
061     * converted into a bit field.
062     */
063    public BitField(byte[] data)
064    {
065        if(null == data){
066            throw new IllegalArgumentException("BitField.BitField((byte[])null) is not allowed.");
067        }
068        if(data.length == 0){
069            throw new IllegalArgumentException("BitField.BitField((byte[]){}) is not allowed.");
070        }
071        m_bits = new boolean [ data.length * Constants.BITS_PER_BYTE ];
072        int idx = 0;
073        for(int i=0; i<data.length; i++){
074            int bits = data[i];
075            int mask = 0x80;
076            for(int j=0; j<Constants.BITS_PER_BYTE; j++){
077                m_bits[idx++] = (bits & mask) != 0;
078                mask >>= 1;
079            }
080        }
081    }
082
083    /**
084     * Method to compare two BitField objects.
085     *
086     * @param other Object to compare this object to
087     *
088     * @return true if the other object is a BitField object with the
089     * same number of bits, and all bits have the same value.  False
090     * otherwise.
091     */
092    public boolean equals(Object other)
093    {
094        // checks type consistency and not-null:
095        if(!(other instanceof BitField)){
096            return false;
097        }
098        BitField that = (BitField)other;
099        return Arrays.equals(m_bits, that.m_bits);
100    }
101
102    /**
103     * @param index value between (0,lenth-1)
104     * @return the bit value at the specified index.
105     */
106    public boolean get(int index)
107    {
108        return m_bits[index];
109    }
110
111    /**
112     * @return the number of bits in the BitField.
113     */
114    public int length()
115    {
116        return m_bits.length;
117    }
118
119    /**
120     * Sets the bit value at the specified index.
121     * @param index value between (0,lenth-1)
122     * @param value boolean value
123     */
124    public void set(int index, boolean value)
125    {
126        m_bits[index] = value;
127    }
128
129    /**
130     * Sets all the bits to the specified value.
131     * @param value the boolean value to be applied to all bits.
132     */
133    public void setAll(boolean value)
134    {
135        for(int i=0; i<m_bits.length; i++){
136            m_bits[i] = value;
137        }
138    }
139
140    /** Sets all the bits to true.
141     */
142    public void setAllTrue()
143    {
144        setAll(true);
145    }
146
147    /** Sets all the bits to true.
148     */
149    public void setAllFalse()
150    {
151        setAll(false);
152    }
153
154    /**
155     * Sets the bit at the specified index to false.
156     * @param index value between (0,length-1)
157     */
158    public void setFalse(int index)
159    {
160        m_bits[index] = false;
161    }
162
163    /**
164     * Sets the bit at the specified index to true.
165     * @param index value between (0,lenth-1)
166     */
167    public void setTrue(int index)
168    {
169        m_bits[index] = true;
170    }
171
172    /** Returns the string representation of the bit field.
173     */
174    public String toString()
175    {
176        StringBuilder sb = new StringBuilder();
177        for(boolean b : m_bits){
178            // put the character at the beginning
179            sb.insert(0, (b ? '1' : '0'));
180        }
181        sb.insert(0, String.format("[%d:", m_bits.length));
182        sb.append(']');
183        return sb.toString();
184    }
185
186    
187}
