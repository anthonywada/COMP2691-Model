001import java.util.Arrays;
002
003/**
004 * Class representing a bit-field of arbitrary (greater than zero)
005 * size.  There are a number of constructors for creating BitField
006 * objects from various other types.  There are a number of toXXX()
007 * methods for getting the value in different types.
008 */
009public class BitField
010{
011    private final boolean[] m_bits;
012
013    /**
014     * Constructor to create a bit-field of the specified
015     * size, with all bits set to zero.
016     *
017     * @param size value greater than zero indicating the number of bits to represent.
018     */
019    public BitField(int size)
020    {
021        if(size <= 0){
022            throw new IllegalArgumentException(String.format("BitField.BitField(%d) is not allowed.", size));
023        }
024        m_bits = new boolean [ size ];
025    }
026
027    /**
028     * Constructor to create a bit field out of a String containing
029     * '1' and '0' characters.  The MSB is assumed to be on the left
030     * size of the string (index zero), with the LSB on the right side
031     * of the string (length - 1).
032     *
033     * In the parsing logic, '0' characters are 'false' and all other
034     * characters are 'true'.
035     *
036     * @param value non-null, non-empty string to be converted to a
037     * byte array.
038     */
039    public BitField(String value)
040    {
041        if(null == value){
042            throw new IllegalArgumentException("BitField.BitField((String)null) is not allowed.");
043        }
044        if(value.length() == 0){
045            throw new IllegalArgumentException("BitField.BitField(\"\") is not allowed.");
046        }
047        m_bits = new boolean [ value.length() ];
048        // change the order
049        for(int i=value.length()-1, idx = 0; i >= 0; i--, idx++){
050            m_bits[idx] = !(value.charAt(i) == '0');
051        }
052    }
053
054    /**
055     * Constructor to create a bit-field from a byte array.  In the
056     * argument byte array, the LSB is (data[0] AND 0x80) and the MSB
057     * is (data[data.length-1] AND 0x1).
058     *
059     * @param data non-null, non-zero-length byte array that is
060     * converted into a bit field.
061     */
062    public BitField(byte[] data)
063    {
064        if(null == data){
065            throw new IllegalArgumentException("BitField.BitField((byte[])null) is not allowed.");
066        }
067        if(data.length == 0){
068            throw new IllegalArgumentException("BitField.BitField((byte[]){}) is not allowed.");
069        }
070        m_bits = new boolean [ data.length * Constants.BITS_PER_BYTE ];
071        int idx = 0;
072        for(int i=0; i<data.length; i++){
073            int bits = data[i];
074            int mask = 0x80;
075            for(int j=0; j<Constants.BITS_PER_BYTE; j++){
076                m_bits[idx++] = (bits & mask) != 0;
077                mask >>= 1;
078            }
079        }
080    }
081
082    /**
083     * Method to compare two BitField objects.
084     *
085     * @param other Object to compare this object to
086     *
087     * @return true if the other object is a BitField object with the
088     * same number of bits, and all bits have the same value.  False
089     * otherwise.
090     */
091    public boolean equals(Object other)
092    {
093        // checks type consistency and not-null:
094        if(!(other instanceof BitField)){
095            return false;
096        }
097        BitField that = (BitField)other;
098        return Arrays.equals(m_bits, that.m_bits);
099    }
100
101    /**
102     * @param index value between (0,lenth-1)
103     * @return the bit value at the specified index.
104     */
105    public boolean get(int index)
106    {
107        return m_bits[index];
108    }
109
110    /**
111     * Sets the bit value at the specified index.
112     * @param index value between (0,lenth-1)
113     * @param value boolean value
114     */
115    public void set(int index, boolean value)
116    {
117        m_bits[index] = value;
118    }
119
120    /**
121     * Sets the bit at the specified index to false.
122     * @param index value between (0,lenth-1)
123     */
124    public void setFalse(int index)
125    {
126        m_bits[index] = false;
127    }
128
129    /**
130     * Sets the bit at the specified index to true.
131     * @param index value between (0,lenth-1)
132     */
133    public void setTrue(int index)
134    {
135        m_bits[index] = true;
136    }
137
138    /**
139     * @return the number of bits in the BitField.
140     */
141    public int size()
142    {
143        return m_bits.length;
144    }
145}
